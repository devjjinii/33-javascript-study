## new, constructor, instanceof, instance


```js
function People(name, age) { // constructor
    this.name  = name;
    this.age = age;
}

let jin = new People('jin', 26); // instance == 객체 (현재생성된)

// constructor 가 있어야 new를 통해 객체를 만들 수 있음. 


//Constructor 함수가 new 연산자를 만나면 instance를 생성.
//instance가 생성되면 Constructor의 프로퍼티인 prototype을 instance의 __proto__에 전달
//prototype이 과 __proto__은 같은 객체를 참조
//__proto__에 접근할 때 생략가능


jin instanceof People; // true
// prototype 객체
```
